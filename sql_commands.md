# database Schema — SQL Commands

This document contains the precise PostgreSQL schema commands reflecting the current Supabase database configuration for the Langchain Email Service Project.

---

## 1. User Settings Table

Tracks user review mode preferences, admin privileges, and history cleanup settings.

```sql
CREATE TABLE IF NOT EXISTS public.user_settings (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid NOT NULL UNIQUE,
  review_mode boolean NOT NULL DEFAULT true,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now(),
  enable_history_cleanup boolean DEFAULT false,
  is_admin boolean DEFAULT false
);

CREATE INDEX IF NOT EXISTS idx_user_settings_user_id ON public.user_settings(user_id);
```

---

## 2. Subscriptions Table

Manages active subscription tiers and monthly limits.

```sql
CREATE TABLE IF NOT EXISTS public.subscriptions (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid NOT NULL UNIQUE,
  tier text NOT NULL DEFAULT 'free',
  emails_used integer DEFAULT 0,
  emails_limit integer DEFAULT 25,
  period_start timestamptz,
  period_end timestamptz,
  status text DEFAULT 'active',
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_subscriptions_user_id ON public.subscriptions(user_id);
```

---

## 3. Gmail OAuth Tokens Table

Stores encrypted Google OAuth credentials.

```sql
CREATE TABLE IF NOT EXISTS public.gmail_oauth_tokens (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid NOT NULL UNIQUE,
  refresh_token text NOT NULL,
  access_token text,
  sender_email text,
  token_expiry timestamptz,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_gmail_oauth_tokens_user_id ON public.gmail_oauth_tokens(user_id);
CREATE INDEX IF NOT EXISTS idx_gmail_oauth_tokens_sender_email ON public.gmail_oauth_tokens(sender_email);
```

---

## 4. Gmail Watch State Table

Keeps track of Gmail history sync cursor ids per user.

```sql
CREATE TABLE IF NOT EXISTS public.gmail_watch_state (
  id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  last_history_id text NOT NULL,
  updated_at timestamptz DEFAULT now(),
  user_id text UNIQUE
);
```

---

## 5. Processed Emails Table

Distributed idempotency mutex lock mapping.

```sql
CREATE TABLE IF NOT EXISTS public.processed_emails (
  id text PRIMARY KEY, -- "user_id:message_id"
  user_id text,
  message_id text,
  status text DEFAULT 'processing',
  created_at timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_processed_emails_created_at ON public.processed_emails(created_at);
```

---

## 6. Processed Messages Table

Historical record of successfully handled Gmail messages.

```sql
CREATE TABLE IF NOT EXISTS public.processed_messages (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  gmail_message_id text UNIQUE,
  processed_at timestamptz DEFAULT now(),
  sender_email text,
  subject text,
  status text
);
```

## 7. Email Filters Table

Holds filter whitelist and blacklist patterns.

```sql
CREATE TABLE IF NOT EXISTS public.email_filters (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid NOT NULL,
  type text NOT NULL, -- 'whitelist' or 'blacklist'
  pattern text NOT NULL,
  created_at timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_email_filters_user_id ON public.email_filters(user_id);
```

## 8. Business Profile Presets Table

Stores system business type configuration presets.

```sql
CREATE TABLE IF NOT EXISTS public.business_profile_presets (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  type_key text NOT NULL UNIQUE,
  name text NOT NULL,
  categories jsonb NOT NULL DEFAULT '[]'::jsonb,
  extraction_fields jsonb NOT NULL DEFAULT '[]'::jsonb,
  default_tone text NOT NULL DEFAULT 'professional',
  default_style text NOT NULL DEFAULT 'concise',
  created_at timestamptz DEFAULT now()
);
```

## 9. User Business Profiles Table

Stores user-customized business tone, style, and category configurations.

```sql
CREATE TABLE IF NOT EXISTS public.user_business_profiles (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid NOT NULL UNIQUE,
  preset_type_key text,
  tone_override text,
  style_override text,
  categories_override jsonb,
  onboarding_complete boolean DEFAULT false,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_user_business_profiles_user_id ON public.user_business_profiles(user_id);
```

---

## 10. Tickets Table

Represents generated customer support tickets.

```sql
CREATE TABLE IF NOT EXISTS public.tickets (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid NOT NULL,
  sender_email text NOT NULL,
  subject text,
  status text NOT NULL DEFAULT 'open',
  created_at timestamptz DEFAULT now(),
  resolved_at timestamptz,
  category text,
  resolution text
);

CREATE INDEX IF NOT EXISTS idx_tickets_user_id ON public.tickets(user_id);
CREATE INDEX IF NOT EXISTS idx_tickets_status ON public.tickets(status);
```

---

## 11. Ticket Messages Table

Tracks discussion history inside support tickets.

```sql
CREATE TABLE IF NOT EXISTS public.ticket_messages (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  ticket_id uuid NOT NULL REFERENCES public.tickets(id) ON DELETE CASCADE,
  direction text NOT NULL,
  body text NOT NULL,
  gmail_message_id text,
  created_at timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_ticket_messages_ticket_id ON public.ticket_messages(ticket_id);
```

---

## 12. Approval Queue Table

Draft AI auto-responses awaiting review.

```sql
CREATE TABLE IF NOT EXISTS public.approval_queue (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  ticket_id uuid REFERENCES public.tickets(id) ON DELETE CASCADE,
  sender_email text NOT NULL,
  reply_subject text NOT NULL,
  original_reply_body text NOT NULL,
  edited_reply_body text,
  status text NOT NULL DEFAULT 'pending',
  created_at timestamptz DEFAULT now(),
  acted_at timestamptz
);

CREATE INDEX IF NOT EXISTS idx_approval_queue_status ON public.approval_queue(status);
CREATE INDEX IF NOT EXISTS idx_approval_queue_ticket_id ON public.approval_queue(ticket_id);
```
